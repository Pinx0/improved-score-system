country_event = 
{
  id = iss_scores.201
  title = "iss_empty_string"
  desc = "iss_empty_string"
  hidden = yes
  is_triggered_only = yes #iss_scores.110
  immediate =
  {
    every_country =
    {
      limit = 
      {
        has_country_flag = iss_show_in_scores_1
      }
      clr_country_flag = iss_show_in_scores_1
    }
    REB = { save_global_event_target_as = iss_country_1 }
    while = 
    {
      limit =
      {
        any_country = 
        {
          OR = 
          {
            was_player = yes
            FROM = { has_global_flag = iss_show_ia_scores }
          }
          check_variable = 
          {
            which = iss_score
            which = event_target:iss_country_1
          }
          NOT =
          {
            OR =
            {
              has_country_flag = iss_show_in_scores_1


              is_variable_equal = 
              {
                which = iss_score
                which = event_target:iss_country_1
              }
            }
          }
        }
      }
      every_country =
      {
        limit = 
        {
          has_country_flag = iss_show_in_scores_1
        }
        clr_country_flag = iss_show_in_scores_1
      }
      random_country = {
        limit =
        {
          OR = 
          {
            was_player = yes
            FROM = { has_global_flag = iss_show_ia_scores }
          }
          check_variable = 
          {
            which = iss_score
            which = event_target:iss_country_1
          }
          NOT =
          {
            OR =
            {
              has_country_flag = iss_show_in_scores_1


              is_variable_equal = 
              {
                which = iss_score
                which = event_target:iss_country_1
              }
            }
          }
        }
        set_country_flag = iss_show_in_scores_1
        save_global_event_target_as = iss_country_1
      }
    }
  }
  option = 
	{
		name = "iss_scores.999.optz"
	}
}
